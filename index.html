闭包 (Closure) —— 函数里面定义函数
def make_counter():              # 外部函数
    count = 0                     # 外部变量(自由变量)
    def counter():                # 内部函数
        nonlocal count           # 声明要修改外部变量
        count += 1                # 每次调用计数+1
        return count
    return counter               # 返回内部函数(带着count变量)

c = make_counter()                # c是一个闭包
print(c())                         # 1
print(c())                         # 2 —— count被"记住"了！

本质：装饰器是一个接受函数作为参数、返回新函数的高阶函数。

Level 1: 最简单的装饰器 —— 计时
import time                         # 导入time模块
def timer(func):                   # 装饰器函数，参数func是被装饰的函数
    def wrapper(*args, **kwargs):   # 包装函数，接收任意参数
        start = time.time()         # 记录开始时间
        result = func(*args, **kwargs) # 执行原函数，保存返回值
        elapsed = time.time() - start # 计算耗时
        print(f"{func.__name__} 耗时: {elapsed:.3f}s")
        return result               # 返回原函数的返回值
    return wrapper                  # 返回包装函数

@timer                              # @timer等同于: slow_func = timer(slow_func)
def slow_func():
    time.sleep(1)                 # 模拟耗时操作
slow_func()                          # 自动计时！
Level 2: 带参数的装饰器 —— 重试机制
def retry(times=3, delay=1):       # 外层函数：接收配置参数
    def decorator(func):           # 中层函数：标准的装饰器
        def wrapper(*args, **kwargs): # 内层函数：包装逻辑
            for i in range(times):    # 循环重试
                try:
                    return func(*args, **kwargs) # 成功就返回
                except Exception as e:   # 捕获异常
                    if i == times - 1:  # 最后一次还是失败
                        raise   # 抛出异常
                    time.sleep(delay)   # 等一会儿再试
        return wrapper            # 返回包装函数
    return decorator              # 返回装饰器

@retry(times=5, delay=0.5)           # 失败重试5次，间隔0.5秒
def call_api(url):               # 调用外部API
    return requests.get(url)
Level 3: @wraps —— 保留原函数元信息
from functools import wraps       # 导入wraps装饰器
def my_decorator(func):
    @wraps(func)                   # 保留func的__name__和__doc__
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper
# 没有@wraps的话，装饰后函数名变成"wrapper"，文档丢失！
🎯 面试这样说：「装饰器本质是闭包+高阶函数。@wraps一定要加，否则函数名和文档会丢。装饰器可以无侵入地添加日志、计时、权限校验、重试等横切功能——这正是AOP面向切面编程的思想。」

生成器：用yield关键字的函数。调用时不立即执行，返回迭代器。每次next()执行到下一个yield并暂停。

# 普通函数 —— 一次返回全部数据(内存杀手)
def get_all_numbers(n):          # 返回n个数字
    result = []                    # 创建一个空列表
    for i in range(n):             # 循环n次
        result.append(i)           # 把每个数字加入列表
    return result                  # 一次返回所有数据 → n=100万时OOM!

# 生成器 —— 一次只产出一个数据(内存友好)
def generate_numbers(n):         # 生成器函数
    for i in range(n):             # 循环n次
        yield i                    # yield"产出"一个值，函数暂停在这

# 使用生成器
gen = generate_numbers(10)       # 调用生成器函数返回一个generator对象
print(next(gen))                   # 0 —— 第一次next，执行到第一个yield
print(next(gen))                   # 1 —— 从上次暂停处继续，到下一个yield

# 生成器表达式 —— 把列表推导式的[]换成()
nums = [x*x for x in range(10_000_000)]  # 列表：占~80MB内存！
nums = (x*x for x in range(10_000_000))  # 生成器：几乎不占内存！

# yield from —— 委托给子生成器(Python 3.3+)
def chain(*iterables):            # 串联多个可迭代对象

def make_counter():              # 外部函数
    count = 0                     # 外部变量(自由变量)
    def counter():                # 内部函数
        nonlocal count           # 声明要修改外部变量

    return counter               # 返回内部函数(带着count变量)

def timer(func):                   # 装饰器函数，参数func是被装饰的函数
    def wrapper(*args, **kwargs):   # 包装函数，接收任意参数
        start = time.time()         # 记录开始时间
        result = func(*args, **kwargs) # 执行原函数，保存返回值
        elapsed = time.time() - start # 计算耗时
        print(f"{func.__name__} 耗时: {elapsed:.3f}s")
        return result               # 返回原函数的返回值
    return wrapper                  # 返回包装函数

@timer                              # @timer等同于: slow_func = timer(slow_func)
def slow_func():
    time.sleep(1)                 # 模拟耗时操作
slow_func()                          # 自动计时！
Level 2: 带参数的装饰器 —— 重试机制
def retry(times=3, delay=1):       # 外层函数：接收配置参数
    def decorator(func):           # 中层函数：标准的装饰器
        def wrapper(*args, **kwargs): # 内层函数：包装逻辑
            for i in range(times):    # 循环重试
                try:
                    return func(*args, **kwargs) # 成功就返回
                except Exception as e:   # 捕获异常
                    if i == times - 1:  # 最后一次还是失败
                        raise   # 抛出异常
                    time.sleep(delay)   # 等一会儿再试
        return wrapper            # 返回包装函数
    return decorator              # 返回装饰器

@retry(times=5, delay=0.5)           # 失败重试5次，间隔0.5秒
def call_api(url):               # 调用外部API
    return requests.get(url)
Level 3: @wraps —— 保留原函数元信息
from functools import wraps       # 导入wraps装饰器
def my_decorator(func):
    @wraps(func)                   # 保留func的__name__和__doc__
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper
# 没有@wraps的话，装饰后函数名变成"wrapper"，文档丢失！
🎯 面试这样说：「装饰器本质是闭包+高阶函数。@wraps一定要加，否则函数名和文档会丢。装饰器可以无侵入地添加日志、计时、权限校验、重试等横切功能——这正是AOP面向切面编程的思想。」

生成器：用yield关键字的函数。调用时不立即执行，返回迭代器。每次next()执行到下一个yield并暂停。

# 普通函数 —— 一次返回全部数据(内存杀手)
def get_all_numbers(n):          # 返回n个数字
    result = []                    # 创建一个空列表
    for i in range(n):             # 循环n次
        result.append(i)           # 把每个数字加入列表
    return result                  # 一次返回所有数据 → n=100万时OOM!

# 生成器 —— 一次只产出一个数据(内存友好)
def generate_numbers(n):         # 生成器函数
    for i in range(n):             # 循环n次
        yield i                    # yield"产出"一个值，函数暂停在这

# 使用生成器
gen = generate_numbers(10)       # 调用生成器函数返回一个generator对象
print(next(gen))                   # 0 —— 第一次next，执行到第一个yield
print(next(gen))                   # 1 —— 从上次暂停处继续，到下一个yield

# 生成器表达式 —— 把列表推导式的[]换成()
nums = [x*x for x in range(10_000_000)]  # 列表：占~80MB内存！
nums = (x*x for x in range(10_000_000))  # 生成器：几乎不占内存！

# yield from —— 委托给子生成器(Python 3.3+)
def chain(*iterables):            # 串联多个可迭代对象
    for it in iterables:
        yield from it            # 等价于 for x in it: yield x
🎯 面试这样说：「处理大数据时能用生成器绝不用列表。生成器核心优势是惰性计算——不一次性加载全部数据，而是按需生成。配合itertools可以实现高效的数据流水线。」

幂等性 —— 腾讯面试隐藏考点

幂等：同一操作执行多次，结果相同。比如支付回调重复推送，不能重复扣款。

🤝 三次握手 — 建立连接
1

SYN (客户端→服务器)

客户端SYN=1, seq=x。"我要连你，这是我的序号"→SYN_SENT状态。

2

SYN+ACK (服务器→客户端)

服务器SYN+ACK=1, seq=y, ack=x+1。"收到，这是我的序号"→SYN_RCVD。

3

ACK (客户端→服务器)

客户端ACK=1, ack=y+1。"收到，连接建立"→ESTABLISHED。

💡 为什么三次？「两次不够——服务器不知道客户端能否收到。三次是验证双向通信能力的最少次数。就像打电话：'喂？'→'喂！'→'好的'。」
👋 四次挥手 — 断开连接
1

FIN (主动→被动)

"我说完了"→FIN_WAIT_1

2

ACK (被动→主动)

"收到，我还有数据"→FIN_WAIT_2 / CLOSE_WAIT

3

FIN (被动→主动)

"我也说完了"→TIME_WAIT(等2MSL)

4

ACK (主动→被动)

"好的"→CLOSED

💡 为什么四次？「TCP是全双工——双方独立收发。关闭需要两个方向各FIN+ACK=四次。」

  • SYN Flood攻击？——伪造大量SYN占满半连接队列。

import re
from collections import defaultdict

def analyze_slow_requests(log_file, threshold=1.0):
    """分析慢请求，返回按次数排序的URL列表"""
    slow_requests = defaultdict(int)
    
    with open(log_file, 'r', encoding='utf-8') as f:
        for line in f:
            # 正则匹配：response_time
            match = re.search(r'"response_time":(\d+\.?\d*)', line)
            if match:
                resp_time = float(match.group(1))
                if resp_time > threshold:
                    # 提取URL
                    url_match = re.search(r'"(GET|POST) (\S+)', line)
                    if url_match:
                        url = url_match.group(2)
                        slow_requests[url] += 1
    
    # 按慢请求次数排序
    return sorted(slow_requests.items(), key=lambda x: x[1], reverse=True)

# 使用
result = analyze_slow_requests('/var/log/nginx/access.log')
for url, count in result[:10]:  # top 10慢接口
    print(f"{url}: {count}次")

# 需求：并发调用API，验证接口的性能
import requests
from concurrent.futures import ThreadPoolExecutor, as_completed
from time import time

def call_api(url, payload):
    """调用单个API，返回状态和耗时"""
    start = time()
    try:
        resp = requests.post(url, json=payload, timeout=5)
        elapsed = time() - start
        return {'status': resp.status_code, 'elapsed': elapsed}
    except Exception as e:
        return {'status': 'ERROR', 'elapsed': time() - start, 'error': str(e)}

def concurrent_test(url, payload, concurrency=10, total=100):
    """并发测试，返回统计结果"""
    results = []
    with ThreadPoolExecutor(max_workers=concurrency) as executor:
        futures = [executor.submit(call_api, url, payload) for _ in range(total)]
        for future in as_completed(futures):
            results.append(future.result())
    
    # 统计
    success = sum(1 for r in results if r['status'] == 200)
    avg_time = sum(r['elapsed'] for r in results) / len(results)
    return {'total': total, 'success': success, 'avg_time': avg_time}

# 使用
result = concurrent_test('http://api.example.com/login', {'user': 'test'}, concurrency=20, total=200)
print(f"成功率: {result['success']/result['total']*100:.1f}%")
print(f"平均响应时间: {result['avg_time']:.3f}s")
🎯 面试亮点：「我用ThreadPoolExecutor写了一个并发测试脚本，可以模拟20/50/100个并发用户，验证接口的性能和稳定性。」

📊 高频问题8：什么是APM？核心指标有哪些？

📊 高频问题9：什么是Trace链路追踪？

标准答案：

「Trace是分布式链路追踪，用来定位跨服务调用的性能瓶颈。
核心概念：
- TraceID：全局唯一，串联整个调用链
- SpanID：每个服务调用的唯一标识
- ParentSpanID：父调用的SpanID

实际案例：
某次玩家登录慢，通过Trace发现是『鉴权服务→数据库』这层慢。
进一步排查发现是数据库连接池满了，优化后P99从2s降到200ms。」
🎯 面试亮点：「我不仅知道Trace概念，还能用Trace定位真实性能问题。」
